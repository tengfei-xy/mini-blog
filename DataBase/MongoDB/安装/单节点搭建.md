```
sudo yum -y install openssl
wget https://fastdl.mongodb.org/linux/mongodb-linux-x86_64-rhel70-5.0.7.tgz
tar zxf mongodb-linux-x86_64-rhel70-5.0.7.tgz
sudo mv mongodb-linux-x86_64-rhel70-5.0.7 /usr/local/mongodb-5.0.7
sudo groupadd mongo
sudo useradd -g mongo -s /bin/bash -p $(openssl passwd -1 "vsU5:d>%PO/2Xv3") -m mongo
sudo chown -R mongo.mongo /usr/local/mongodb-5.0.7/
```

```
su - mongo
```

```
cd /usr/local/mongodb-5.0.7/
rm -rf  LICENSE-Community.txt MPL-2 README THIRD-PARTY-NOTICES
mkdir log etc data
vi etc/mongod.conf
```

启动

```
bin/mongod -f etc/mongod.conf
```

```
bin/mongo 127.0.0.1:28018
```

创建管理员

```
use admin;
db.createUser({user: "root", pwd: "666", roles: [{ role: "root", db: "admin" }]})
exit
```

创建完成后重新进入控制台

```
bin/mongo 127.0.0.1:28018/admin -u root -p
```

use gsri;
