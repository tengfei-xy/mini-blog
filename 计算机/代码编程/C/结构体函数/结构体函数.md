# 结构体函数

```c
#include <stdio.h>

// 定义person结构体类型
typedef struct person {
    char *name; // 姓名
    void (*say_hello)(struct person *); // 函数指针
} person;

// 定义一个打招呼的函数
void hello(person *p) {
    printf("Hello, I'm %s.\n", p->name);
}

int main() {
    // 定义并初始化tom变量
    person tom = {"Tom", hello};
    // 通过tom调用say_hello函数
    tom.say_hello(&tom); // 输出：Hello, I'm Tom.
    return 0;
}
```
