```bash
# 安装命令
bash <(curl -sL https://raw.githubusercontent.com/hijkpw/scripts/master/goV2.sh)

# 查看端口
cat /etc/v2ray/config.json

# 开放端口
firewall-cmd --permanent --add-port=30749/tcp --zone=xunyang
firewall-cmd --reload

# 开机自启
systemctl enable v2ray
systemctl start v2ray

# 可能需要禁用SELinux
setenforce 0
```
