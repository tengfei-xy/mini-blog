获取格式化时间

```c
#include <time.h>

time_t t;
time(&t);
struct tm * timeinfo = localtime(&t);
char * strtime = asctime(timeinfo);
strtime[strlen(strtime)-1] = 32;
printf("%s",strtime);
```

获取当前时间戳

```c
#include <time.h>

time_t t= time(NULL);
printf("timestamp=%d", time(&t));
```

获取毫秒时间戳

```c
#include <time.h>
#include <sys/timeb.h>

struct timeb st;
ftime(&st);
time_t t= time(NULL);
printf("millitm_timestamp=%ld",time(&t)*1000+st.millitm);
```
